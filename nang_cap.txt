@app.route("/voice", methods=["POST"])
def voice():
    text = request.json["text"]

    intent = detect_intent(text)

    if intent["type"] == "call":
        room_id = create_room()
        send_push(intent["target"], room_id)

        return {
            "action": "calling",
            "message": "Đang gọi người thân giúp bác",
            "room": room_id
        }

    return {"action": "chat"}
2.3 Detect intent gọi
contacts = {
    "con trai": "user_2",
    "con gái": "user_3"
}

def detect_intent(text):
    text = text.lower()

    if "gọi" in text:
        for name in contacts:
            if name in text:
                return {
                    "type": "call",
                    "target": contacts[name]
                }

    return {"type": "chat"}

import requests

def send_push(user_id, room_id):
    token = get_user_token(user_id)

    requests.post(
        "https://fcm.googleapis.com/fcm/send",
        headers={
            "Authorization": "key=YOUR_SERVER_KEY",
            "Content-Type": "application/json"
        },
        json={
            "to": token,
            "notification": {
                "title": "Mẹ đang gọi",
                "body": "Nhấn để trả lời"
            },
            "data": {
                "room": room_id
            }
        }
    )
