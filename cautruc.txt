2. Backend Flask (app.py)

Phiên bản rút gọn nhưng đủ dùng (có Gemini + memory + streaming cơ bản)

import google.generativeai as genai
from flask import Flask, request, jsonify, Response, render_template

app = Flask(__name__)

genai.configure(api_key="YOUR_API_KEY")
model = genai.GenerativeModel("gemini-2.5-pro")

# load knowledge
with open("chatbot.txt", "r", encoding="utf-8") as f:
    knowledge = f.read()

chat_history = []

def build_prompt(question):
    history = "\n".join(chat_history[-6:])

    return f"""
Bạn là trợ lý thân thiện dành cho người cao tuổi.

- Nói chậm, dễ hiểu
- Giọng ấm áp
- Tránh từ khó

Lịch sử:
{history}

Thông tin:
{knowledge}

Người dùng:
{question}

Trả lời:
"""

@app.route("/")
def index():
    return render_template("index.html")

@app.route("/chat_stream", methods=["POST"])
def chat_stream():
    user_text = request.json["message"]
    prompt = build_prompt(user_text)

    def generate():
        response = model.generate_content(prompt, stream=True)
        full = ""

        for chunk in response:
            if chunk.text:
                full += chunk.text
                yield chunk.text

        chat_history.append(f"User: {user_text}")
        chat_history.append(f"Bot: {full}")

    return Response(generate(), mimetype="text/plain")

if __name__ == "__main__":
    app.run()

4. JavaScript (voice + streaming)
function startVoice() {
  const recognition = new webkitSpeechRecognition();
  recognition.lang = "vi-VN";

  recognition.onresult = function(e) {
    const text = e.results[0][0].transcript;
    sendToServer(text);
  };

  recognition.start();
}

async function sendToServer(text) {
  const res = await fetch("/chat_stream", {
    method: "POST",
    headers: {"Content-Type": "application/json"},
    body: JSON.stringify({message: text})
  });

  const reader = res.body.getReader();
  const decoder = new TextDecoder();

  let full = "";

  while (true) {
    const {done, value} = await reader.read();
    if (done) break;

    const chunk = decoder.decode(value);
    full += chunk;

    document.getElementById("output").innerText = full;
  }

  speak(full);
}

function speak(text) {
  const speech = new SpeechSynthesisUtterance(text);
  speech.lang = "vi-VN";
  speech.rate = 0.85;
  speechSynthesis.speak(speech);
}

service worker (sw.js)
self.addEventListener("install", event => {
  event.waitUntil(
    caches.open("cache").then(cache => {
      return cache.addAll([
        "/",
        "/static/script.js"
      ]);
    })
  );
});

self.addEventListener("fetch", event => {
  event.respondWith(
    caches.match(event.request).then(res => {
      return res || fetch(event.request);
    })
  );
});
