from flask import Response

@app.route("/chat_stream", methods=["POST"])
def chat_stream():
    user_text = request.json["message"]
    prompt = build_prompt(user_text)

    def generate():
        response = model.generate_content(prompt, stream=True)

        full_text = ""

        for chunk in response:
            if chunk.text:
                full_text += chunk.text
                yield chunk.text

        # lưu history sau khi xong
        chat_history.append(f"User: {user_text}")
        chat_history.append(f"Bot: {full_text}")

    return Response(generate(), mimetype="text/plain")
Frontend nhận stream
async function sendToServer(text) {
  const response = await fetch("/chat_stream", {
    method: "POST",
    headers: {"Content-Type": "application/json"},
    body: JSON.stringify({message: text})
  });

  const reader = response.body.getReader();
  const decoder = new TextDecoder();

  let fullText = "";

  while (true) {
    const { done, value } = await reader.read();
    if (done) break;

    const chunk = decoder.decode(value);
    fullText += chunk;

    updateUI(fullText); // hiển thị dần
  }

  speak(fullText);
}

3.1 Prompt nâng cấp (rất quan trọng)
Bạn là một người trợ lý giống như người thân trong gia đình, chuyên trò chuyện với người lớn tuổi.

Cách nói:
- Ngắn gọn, rõ ràng
- Giải thích đơn giản
- Giọng nhẹ nhàng, ấm áp
- Có thể dùng câu như: "dạ", "ạ", "vâng"

Hành vi:
- Nếu người dùng hỏi sức khỏe → quan tâm
- Nếu người dùng hỏi thông tin → giải thích dễ hiểu
- Nếu người dùng nói chuyện → phản hồi tự nhiên như người thật

Không được:
- Trả lời quá dài
- Dùng từ chuyên môn khó hiểu

3.2 Chia nhỏ câu để đọc (giống người thật)
function speak(text) {
  const sentences = text.split(/[.!?]/);

  let i = 0;

  function speakNext() {
    if (i >= sentences.length) return;

    const utter = new SpeechSynthesisUtterance(sentences[i]);
    utter.lang = "vi-VN";
    utter.rate = 0.85;

    utter.onend = () => {
      i++;
      speakNext();
    };

    speechSynthesis.speak(utter);
  }

  speakNext();
}

    return "\n".join([
        c for c in chunks if any(word in c.lower() for word in question.lower().split())
